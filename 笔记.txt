1、Vue使用Element组件：
①npm i element-ui -S
②在 main.js 中写入以下内容：(去掉重复的内容)
import Vue from 'vue'
import ElementUI from 'element-ui'
import 'element-ui/lib/theme-chalk/index.css'
import App from './App.vue'

Vue.use(ElementUI)

new Vue({
  el: '#app',
  render: h => h(App)
})
③测试使用：
<el-button>test</el-button>
